     3	import 'package:drift/drift.dart';
     4	import 'package:tenniscourtcare/data/database/app_database.dart' as db;
     5	import 'package:tenniscourtcare/data/models/app_event_model.dart';
     6	import 'package:tenniscourtcare/domain/entities/app_event.dart';
     7	import 'package:tenniscourtcare/domain/entities/sync_status.dart';
     8
     9	class EventMapper {
    10	  // Model → Domain Entity
    11	  static AppEvent toDomain(AppEventModel model) {
    12	    return AppEvent(
    13	      id: model.id,
    14	      title: model.title,
    15	      description: model.description,
    16	      startTime: DateTime.parse(model.startTime),
    17	      endTime: DateTime.parse(model.endTime),
    18	      color: model.color,
    19	      terrainIds: model.terrainIds,
    20	      syncStatus: SyncStatus.fromString(model.syncStatus),
    21	      createdAt: DateTime.parse(model.createdAt),
    22	      updatedAt: DateTime.parse(model.updatedAt),
    23	      firebaseId: model.firebaseId,
    24	      createdBy: model.createdBy,
    25	      modifiedBy: model.modifiedBy,
    26	    );
    27	  }
    28
    29	  // Domain Entity → Model
    30	  static AppEventModel toModel(AppEvent domain) {
    31	    return AppEventModel(
    32	      id: domain.id,
    33	      title: domain.title,
    34	      description: domain.description,
    35	      startTime: domain.startTime.toIso8601String(),
    36	      endTime: domain.endTime.toIso8601String(),
    37	      color: domain.color,
    38	      terrainIds: domain.terrainIds,
    39	      syncStatus: domain.syncStatus.name,
    40	      createdAt: domain.createdAt.toIso8601String(),
    41	      updatedAt: domain.updatedAt.toIso8601String(),
    42	      firebaseId: domain.firebaseId,
    43	      createdBy: domain.createdBy,
    44	      modifiedBy: domain.modifiedBy,
    45	    );
    46	  }
    47
    48	  // Drift Entity → Domain Entity
    49	  static AppEvent fromDriftEntity(db.EventRow driftEntity) {
    50	    return AppEvent(
    51	      id: driftEntity.id ?? 0,
    52	      title: driftEntity.title,
    53	      description: driftEntity.description,
    54	      startTime: driftEntity.startTime,
    55	      endTime: driftEntity.endTime,
    56	      color: driftEntity.color,
    57	      terrainIds: driftEntity.terrainIds,
    58	      syncStatus: SyncStatus.fromString(driftEntity.syncStatus),
    59	      createdAt: driftEntity.createdAt ?? DateTime.now(),
    60	      updatedAt: driftEntity.updatedAt ?? DateTime.now(),
    61	      firebaseId: driftEntity.firebaseId,
    62	      createdBy: driftEntity.createdBy,
    63	      modifiedBy: driftEntity.modifiedBy,
    64	    );
    65	  }
    66	}
    67
    68	// Extensions for compatibility
    69	extension AppEventModelX on AppEventModel {
    70	  AppEvent toDomain() => EventMapper.toDomain(this);
    71	}
    72
    73	extension EventDriftX on db.EventRow {
    74	  AppEvent toDomain() => EventMapper.fromDriftEntity(this);
    75	}
    76
    77	// Domain → Companion (Keeping for DB inserts)
    78	extension AppEventDomainX on AppEvent {
    79	  db.EventsCompanion toCompanion({bool includeId = true}) => db.EventsCompanion(
    80	    id: includeId && id != null ? Value(id!) : const Value.absent(),
    81	    title: Value(title),
    82	    description: Value(description),
    83	    startTime: Value(startTime),
    84	    endTime: Value(endTime),
    85	    color: Value(color),
    86	    terrainIds: Value(terrainIds),
    87	    // Sync mappings
    88	    syncStatus: Value(syncStatus.name),
    89	    createdAt: Value(createdAt),
    90	    updatedAt: Value(updatedAt),
    91	    firebaseId: Value(firebaseId),
    92	    createdBy: Value(createdBy),
    93	    modifiedBy: Value(modifiedBy),
    94	  );
    95	}
